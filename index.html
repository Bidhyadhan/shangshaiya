const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); require('dotenv').config(); const app = express(); app.use(cors()); app.use(express.json()); // 1. DATABASE CONNECTION mongoose.connect(process.env.MONGODB_URI) .then(() => console.log("✨ Connected to SHANGSHAIYA MongoDB")) .catch(err => console.error("❌ Connection error:", err)); // 2. PRODUCT SCHEMA const productSchema = new mongoose.Schema({ id: { type: String, unique: true }, name: String, price: Number, type: String, stock: { type: Number, default: 15 }, visible: { type: Boolean, default: true }, artisan: String, story: String, image: String }); const Product = mongoose.model('Product', productSchema); // 3. API ROUTES app.get('/api/products', async (req, res) => { const products = await Product.find({}); res.json(products); }); // Update stock on purchase app.patch('/api/products/:id/buy', async (req, res) => { const product = await Product.findOneAndUpdate( { id: req.params.id, stock: { $gt: 0 } }, { $inc: { stock: -1 } }, { new: true } ); res.json(product || { error: "Out of stock" }); }); // Admin update (Price/Visibility) app.put('/api/products/:id', async (req, res) => { const updated = await Product.findOneAndUpdate({ id: req.params.id }, req.body, { new: true }); res.json(updated); }); app.listen(5000, () => console.log("🚀 Server: http://localhost:5000"));
